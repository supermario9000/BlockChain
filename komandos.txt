const c = await OrderFulfillment.deployed();
await c.createOrder()
const me = (await web3.eth.getAccounts())[0];

const balWei = await web3.eth.getBalance(me);
web3.utils.fromWei(balWei, 'ether');

const tx = await c.createOrder({ from: me });
const id = tx.logs[0].args.orderId.toNumber();

const f = web3.utils.toWei('0.001', 'ether');
const s = web3.utils.toWei('0.001', 'ether');
await c.setPrice(id, f, s, { from: me });

await c.markProcessing(id, { from: me });
await c.requestPayment(id, { from: me });

const total = web3.utils.toBN(f).add(web3.utils.toBN(s));
await c.pay(id, { from: me, value: total });

await c.uploadInvoice(id, 'ipfs://QmXxxx', { from: me });
await c.closeAndPayout(id, { from: me });
